/**
 * This class serves as a hash table of string-any value pairs. It is also capable of holding native D objects.
 * Objects may have a __proto__ field that is recursively searched when dictionary look ups fail. When an
 * object is created from the new operator, the "prototype" field of the function becomes the __proto__ field
 * of the created object, much like JavaScript.
 */
class Object {
    /**
     * Creates a new empty object whose __proto__ is the parameter.
     * @param {object} object The object to serve as the new object's __proto__
     * @returns {object} A newly created object with the __proto__ set to the parameter.
     */
    static create(object) {}

    /**
     * Get the key-value entries of an object
     * @param {object} object The object to get the key-value entries from
     * @returns {array} An array of length-2 arrays whose first element is the key and second element associated value.
     */
    static entries(object) {}

    /**
     * Look for a specific property on the object without searching the prototype chain.
     * @param {object} object The object to search
     * @param {string} propName The name of the property
     * @returns {object} an object containing a possible get and set field representing the functions of the property.
     */
    static getOwnPropertyDescriptor(object, propName) {}

    /**
     * Return an array of all the keys in the specific object. Does not recursively search prototype chain.
     * @param {object} object The object whose keys to get
     * @returns {array} An array of all string keys in no specific order
     */
    static keys(object) {}

    /**
     * Returns an array of all values contained in an object
     * @param {object} object The object whose values to return
     * @returns {array} An array of values in no specific order 
     */
    static values(object) {}
}
